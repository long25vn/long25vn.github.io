#include <stdio.h>
#include <math.h>
float giaithua(float x)
{
   if(x <= 1)
   {
      return 1;
   }
   return x * giaithua(x - 1);
}
float luythua(float a, float b)
{   if  (b==0) return 1;
    if (b==1) return a;

    return a*luythua(a,b-1);
}
float sinx(float x, float n)
{   float result=0;
    x = x * (3.1415926 / 180.0);
    for(int i=0;i<=n;i++)
    {
        float z = (2*i)+1;
        result=result+ (luythua(-1,i)*(luythua(x,z) / giaithua(z)));
    }
    return result;
}

    float x;
    float n = 1000;
    printf("Tinh sin cua: ");
    scanf("%f",&x);
    printf("\nSin cua %f bang :%f",x,sinx(x,n));
}
