#include<stdio.h>
void hinhvuongdac (int b)
{
    printf("Nhap do dai canh hinh vuong: ");
    scanf("%d", &b);
    for ( int i = 1; i <= b; i++)
    {
        for (int j = 1; j <= b; j++)
            printf(" * ");
        printf("\n\n");
    }
}
void hinhvuongrong (int c)
{
    printf("nhap do dai canh cua hinh vuong   ");
    scanf("%d", &c);
    if (c < 3)
    {
        printf("Khong the in hinh.");
    }
    else
    {
        for ( int i = 1; i <= c; i++)
        {
            for (int j = 1; j <= c; j++)
                if (i == 1 || j == 1 || i == c || j == c)
                {
                    printf(" * ");
                }
                else
                {
                    printf("   ");
                }
            printf("\n\n");
        }
    }
}
void hinhvuongcheo(int d)
{
    printf("nhap do dai canh cua hinh vuong");
    scanf("%d", &d);
    if (d % 2 == 0 || d < 5)
    {
        printf("Khong the in hinh.");
    }
    else
    {
        for ( int i = 1; i <= d; i++)
        {
            for ( int j = 1; j <= d; j++)
                if (i == 1 || j == 1 || i == d || j == d || i == j || (i + j) == (d + 1))
                {
                    printf(" * ");
                }
                else
                {
                    printf("   ");
                }
            printf("\n\n");
        }
    }
}

}
void hinhvuongthoi(int f)
{
    printf("nhap do dai canh hinh vuong");
    scanf("%d", &f);
    for ( int i = 1; i <= (f + 1) / 2; i++)
    {
        for (int j = 1; j <= f; j++)
            if (i + j == (f + 1) / 2 + 1 || j - i == (f + 1) / 2 - 1 || j == 1 || j == f || i == 1)
                printf(" * ");
            else
                printf("   ");
        printf("\n\n");
    }
    for (int i = 0; i < (f + 1) / 2; i++)
    {
        for (int j = 0; j < f; j++)
        {
            if (j == 0 || i == (f + 1) / 2 - 1 || (i + j) == f - 1 || j == f - 1 || i == j)
                printf(" * ");
            else
                printf("   ");
        }
        printf("\n\n");
    }
}
